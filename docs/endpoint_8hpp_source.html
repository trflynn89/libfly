188 Endpoint<IPAddressType>::from_string(std::string_view endpoint)
189 {
190  constexpr const auto s_max16 =
191  static_cast<std::size_t>(std::numeric_limits<std::uint16_t>::max());
192  constexpr const auto s_colon = ':';
193 
194  const std::size_t separator = endpoint.find_last_of(s_colon);
195 
196  if ((separator == std::string_view::npos) || (separator == 0) || (separator == endpoint.size()))
197  {
198  return std::nullopt;
199  }
200 
201  auto address_view = endpoint.substr(0, separator);
202  auto port_view = endpoint.substr(separator + 1);
203 
204  if constexpr (is_ipv6())
205  {
206  constexpr const auto s_left_bracket = '[';
207  constexpr const auto s_right_bracket = ']';
208 
209  if (address_view.starts_with(s_left_bracket) && address_view.ends_with(s_right_bracket))
210  {
211  address_view = address_view.substr(1, address_view.size() - 2);
212  }
213  else
214  {
215  return std::nullopt;
216  }
217  }
218 
219  fly::BasicStringLexer<std::string> lexer(std::move(port_view));
220 
221  auto address = IPAddressType::from_string(std::move(address_view));
222  auto port = lexer.consume_number();
223 
224  if (!address || !port || (*port > s_max16) || lexer.peek())
225  {
226  return std::nullopt;
227  }
228 
229  return Endpoint(*std::move(address), static_cast<port_type>(*port));
230 }
