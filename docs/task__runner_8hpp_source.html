685 template <typename TaskType, typename ReplyType>
686 Task TaskRunner::wrap_task(TaskType &&task, ReplyType &&reply)
687 {
688  static_assert(std::is_invocable_v<TaskType>, "Task must be invocable without any arguments");
689 
690  using ResultType = std::invoke_result_t<TaskType>;
691  static constexpr bool s_result_is_void = std::is_void_v<ResultType>;
692 
693  static_assert(
694  (s_result_is_void && std::is_invocable_v<ReplyType>) ||
695  (!s_result_is_void && std::is_invocable_v<ReplyType, ResultType>),
696  "Either the task must return a non-void type and the reply must be invocable with only "
697  "that type, or the task must return void and the reply must be invocable without any "
698  "arguments");
699 
700  TaskHolder<TaskType> task_holder {std::forward<TaskType>(task)};
701  TaskHolder<ReplyType> reply_holder {std::forward<ReplyType>(reply)};
702 
703  return
704  [task_holder = std::move(task_holder),
705  reply_holder = std::move(reply_holder)](TaskRunner *runner, TaskLocation location) mutable
706  {
707  if constexpr (s_result_is_void)
708  {
709  std::invoke(std::move(task_holder.m_task));
710  runner->post_task(std::move(location), std::move(reply_holder.m_task));
711  }
712  else
713  {
714  auto result = std::invoke(std::move(task_holder.m_task));
715 
716  runner->post_task(
717  std::move(location),
718  std::bind(std::move(reply_holder.m_task), std::move(result)));
719  }
720  };
721 }
