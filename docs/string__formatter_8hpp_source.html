255 template <typename StringType, typename... ParameterTypes>
256 StringType BasicStringFormatter<StringType, ParameterTypes...>::format(FormatString &&fmt)
257 {
258  auto formatter = [this](auto &&specifier, const auto &value)
259  {
260  this->format_value(std::move(specifier), value);
261  };
262 
263  const view_type view = fmt.view();
264  m_buffer.reserve(view.size() * 2);
265 
266  for (std::size_t pos = 0; pos < view.size();)
267  {
268  switch (const auto &ch = view[pos])
269  {
270  case s_left_brace:
271  if (view[pos + 1] == s_left_brace)
272  {
273  m_buffer.push_back(ch);
274  pos += 2;
275  }
276  else
277  {
278  auto specifier = *std::move(fmt.next_specifier());
279  pos += specifier.m_size;
280 
281  m_parameters.visit(std::move(specifier), formatter);
282  }
283  break;
284 
285  case s_right_brace:
286  m_buffer.push_back(ch);
287  pos += 2;
288  break;
289 
290  default:
291  m_buffer.push_back(ch);
292  ++pos;
293  break;
294  }
295  }
296 
297  return std::move(m_buffer);
298 }

341 template <typename StringType, typename... ParameterTypes>
342 template <typename T, fly::enable_if<detail::is_like_supported_string<T>>>
343 inline void BasicStringFormatter<StringType, ParameterTypes...>::format_value(
344  FormatSpecifier &&specifier,
345  const T &value)
346 {
347  using string_like_type = detail::is_like_supported_string_t<T>;
348 
349  const std::size_t min_width = resolve_size(specifier.m_width);
350  const std::size_t max_width = resolve_size(specifier.m_precision, StringType::npos);
351 
352  const std::size_t actual_size = BasicStringClassifier<string_like_type>::size(value);
353  const std::size_t value_size = std::min(max_width, actual_size);
354 
355  const std::size_t padding_size = std::max(value_size, min_width) - value_size;
356  const auto padding_char = specifier.m_fill ? *specifier.m_fill : s_space;
357 
358  switch (specifier.m_alignment)
359  {
360  case FormatSpecifier::Alignment::Left:
361  case FormatSpecifier::Alignment::Default:
362  append_string(value, value_size);
363  m_buffer.append(padding_size, padding_char);
364  break;
365 
366  case FormatSpecifier::Alignment::Right:
367  m_buffer.append(padding_size, padding_char);
368  append_string(value, value_size);
369  break;
370 
371  case FormatSpecifier::Alignment::Center:
372  {
373  const std::size_t left_padding = padding_size / 2;
374  const std::size_t right_padding =
375  (padding_size % 2 == 0) ? left_padding : left_padding + 1;
376 
377  m_buffer.append(left_padding, padding_char);
378  append_string(value, value_size);
379  m_buffer.append(right_padding, padding_char);
380  break;
381  }
382  }
383 }

387 template <typename T, fly::enable_if<BasicFormatTraits::is_integral<T>>>
388 inline void BasicStringFormatter<StringType, ParameterTypes...>::format_value(
389  FormatSpecifier &&specifier,
390  T value)
391 {
392  if constexpr (std::is_signed_v<T>)
393  {
394  using U = std::make_unsigned_t<std::remove_cvref_t<T>>;
395 
396  // Compute the absolute value of the integer. Benchmarks showed this is exactly as fast as
397  // std::abs, but this also tracks whether the original value was negative without branches.
398  const T sign = value >> std::numeric_limits<T>::digits;
399  const U unsigned_value = static_cast<U>(static_cast<U>(value ^ sign) + (sign & 1));
400 
401  format_value(std::move(specifier), unsigned_value, static_cast<bool>(sign));
402  }
403  else
404  {
405  format_value(std::move(specifier), value, false);
406  }
407 }

410 template <typename StringType, typename... ParameterTypes>
411 template <typename T, fly::enable_if<BasicFormatTraits::is_integral<T>>>
412 void BasicStringFormatter<StringType, ParameterTypes...>::format_value(
413  FormatSpecifier &&specifier,
414  T value,
415  bool is_negative)
416 {
417  static_assert(std::is_unsigned_v<T>);
418 
419  const std::size_t original_size = m_buffer.size();
420  std::size_t prefix_size = 0;
421  std::size_t value_size = 0;
422 
423  if (specifier.m_type == FormatSpecifier::Type::Character)
424  {
425  static constexpr T s_max = static_cast<T>(std::numeric_limits<char_type>::max());
426 
427  if (!is_negative && (value <= s_max))
428  {
429  m_buffer.push_back(static_cast<char_type>(value));
430  value_size = 1;
431  }
432  }
433  else
434  {
435  const bool is_upper_case = specifier.m_case == FormatSpecifier::Case::Upper;
436  const int base = static_cast<int>(specifier.m_type);
437 
438  if (is_negative)
439  {
440  m_buffer.push_back(s_minus_sign);
441  }
442  else if (specifier.m_sign == FormatSpecifier::Sign::Always)
443  {
444  m_buffer.push_back(s_plus_sign);
445  }
446  else if (specifier.m_sign == FormatSpecifier::Sign::NegativeOnlyWithPositivePadding)
447  {
448  m_buffer.push_back(s_space);
449  }
450 
451  if (specifier.m_alternate_form)
452  {
453  m_buffer.push_back(s_zero);
454 
455  if (specifier.m_type == FormatSpecifier::Type::Binary)
456  {
457  m_buffer.push_back(is_upper_case ? s_upper_b : s_lower_b);
458  }
459  else if (specifier.m_type == FormatSpecifier::Type::Hex)
460  {
461  m_buffer.push_back(is_upper_case ? s_upper_x : s_lower_x);
462  }
463  }
464 
465  prefix_size = m_buffer.size() - original_size;
466  value_size = append_number(value, base) + prefix_size;
467 
468  if ((specifier.m_type == FormatSpecifier::Type::Hex) && is_upper_case)
469  {
470  for (std::size_t i = original_size + prefix_size; i < m_buffer.size(); ++i)
471  {
472  m_buffer[i] = classifier::to_upper(m_buffer[i]);
473  }
474  }
475  }
476 
477  const std::size_t width = resolve_size(specifier.m_width);
478  const std::size_t padding_size = std::max(value_size, width) - value_size;
479  const char_type padding_char = specifier.m_fill ? *specifier.m_fill : s_space;
480 
481  switch (specifier.m_alignment)
482  {
483  case FormatSpecifier::Alignment::Left:
484  m_buffer.append(padding_size, padding_char);
485  break;
486 
487  case FormatSpecifier::Alignment::Right:
488  m_buffer.insert(original_size, padding_size, padding_char);
489  break;
490 
491  case FormatSpecifier::Alignment::Center:
492  {
493  const std::size_t left_padding = padding_size / 2;
494  const std::size_t right_padding =
495  (padding_size % 2 == 0) ? left_padding : left_padding + 1;
496 
497  m_buffer.insert(original_size, left_padding, padding_char);
498  m_buffer.append(right_padding, padding_char);
499  break;
500  }
501 
502  case FormatSpecifier::Alignment::Default:
503  if (specifier.m_zero_padding)
504  {
505  m_buffer.insert(original_size + prefix_size, padding_size, s_zero);
506  }
507  else
508  {
509  m_buffer.insert(original_size, padding_size, padding_char);
510  }
511  break;
512  }
513 }

517 template <typename T, fly::enable_if<std::is_floating_point<T>>>
518 inline void BasicStringFormatter<StringType, ParameterTypes...>::format_value(
519  FormatSpecifier &&specifier,
520  const T &value)
521 {
522  static thread_local std::stringstream s_stream;
523  stream_modifiers modifiers(s_stream);
524 
525  if (specifier.m_alignment == FormatSpecifier::Alignment::Default)
526  {
527  specifier.m_alignment = FormatSpecifier::Alignment::Right;
528  }
529 
530  switch (specifier.m_sign)
531  {
532  case FormatSpecifier::Sign::Always:
533  modifiers.setf(std::ios_base::showpos);
534  break;
535 
536  case FormatSpecifier::Sign::NegativeOnlyWithPositivePadding:
537  modifiers.template locale<PositivePaddingFacet<char>>();
538  modifiers.setf(std::ios_base::showpos);
539  break;
540 
541  default:
542  break;
543  }
544 
545  if (specifier.m_alternate_form)
546  {
547  modifiers.setf(std::ios_base::showpoint);
548  }
549 
550  if (specifier.m_zero_padding)
551  {
552  modifiers.setf(std::ios_base::internal, std::ios_base::adjustfield);
553  modifiers.fill(static_cast<char>(s_zero));
554  modifiers.width(resolve_size<std::streamsize>(specifier.m_width));
555  }
556 
557  modifiers.precision(resolve_size<std::streamsize>(specifier.m_precision, 6));
558  specifier.m_precision = std::nullopt;
559 
560  switch (specifier.m_type)
561  {
562  case FormatSpecifier::Type::HexFloat:
563  modifiers.setf(std::ios_base::fixed | std::ios_base::scientific);
564  break;
565 
566  case FormatSpecifier::Type::Scientific:
567  modifiers.setf(std::ios_base::scientific, std::ios::floatfield);
568  break;
569 
570  case FormatSpecifier::Type::Fixed:
571  // Only Apple's Clang seems to respect std::uppercase with std::fixed values. To
572  // ensure consistency, format these values as general types.
573  if (!std::isnan(value) && !std::isinf(value))
574  {
575  modifiers.setf(std::ios_base::fixed, std::ios::floatfield);
576  }
577  break;
578 
579  default:
580  break;
581  }
582 
583  if (specifier.m_case == FormatSpecifier::Case::Upper)
584  {
585  modifiers.setf(std::ios_base::uppercase);
586  }
587 
588  s_stream << value;
589  format_value(std::move(specifier), s_stream.str());
590  s_stream.str({});
591 }

622 template <typename T, fly::enable_if<detail::is_like_supported_string<T>>>
623 void BasicStringFormatter<StringType, ParameterTypes...>::append_string(
624  const T &value,
625  std::size_t value_size)
626 {
627  using string_like_type = detail::is_like_supported_string_t<T>;
628  using view_like_type = std::basic_string_view<typename string_like_type::value_type>;
629  view_like_type view;
630 
631  if constexpr (std::is_array_v<T> || std::is_pointer_v<T>)
632  {
633  view = view_like_type(value, value_size);
634  }
635  else
636  {
637  view = view_like_type(value).substr(0, value_size);
638  }
639 
640  if constexpr (std::is_same_v<StringType, string_like_type>)
641  {
642  m_buffer.append(view.data(), view.size());
643  }
644  else
645  {
646  using unicode = BasicStringUnicode<string_like_type>;
647 
648  const std::size_t original_size = m_buffer.size();
649  auto out = std::back_inserter(m_buffer);
650 
651  if (!unicode::template convert_encoding_into<StringType>(view, out))
652  {
653  m_buffer.resize(original_size);
654  }

659 template <typename StringType, typename... ParameterTypes>
660 template <typename T, fly::enable_if<BasicFormatTraits::is_integral<T>>>
661 std::size_t BasicStringFormatter<StringType, ParameterTypes...>::append_number(T value, int base)
662 {
663  const std::size_t digits = count_digits(value, base);
664 
665  if constexpr (std::is_same_v<StringType, std::string>)
666  {
667  m_buffer.resize(m_buffer.size() + digits);
668 
669  char *end = m_buffer.data() + m_buffer.size();
670  char *begin = end - digits;
671 
672  std::to_chars(begin, end, value, base);
673  }
674  else
675  {
676  static thread_local std::array<char, std::numeric_limits<std::uintmax_t>::digits> s_buffer;
677  using unicode = BasicStringUnicode<std::string>;
678 
679  char *begin = s_buffer.data();
680  char *end = begin + s_buffer.size();
681 
682  const auto result = std::to_chars(begin, end, value, base);
683  std::string_view view(begin, static_cast<std::size_t>(std::distance(begin, result.ptr)));
684 
685  unicode::template convert_encoding_into<StringType>(view, std::back_inserter(m_buffer));
686  }
687 
688  return digits;
689 }
