1675 template <typename T, enable_if<JsonTraits::is_boolean<T>>>
1676 Json::operator T() const noexcept
1677 {
1678  auto visitor = [](const auto &storage) noexcept -> T
1679  {
1680  using S = std::decay_t<decltype(storage)>;
1681 
1682  if constexpr (JsonTraits::is_container_v<S>)
1683  {
1684  return !storage.empty();
1685  }
1686  else if constexpr (JsonTraits::is_floating_point_v<S>)
1687  {

1693  }
1694  else
1695  {
