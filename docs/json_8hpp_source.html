1639 Json::operator T() const &noexcept(false)
1640 {
1641  auto visitor = [this](const auto &storage) -> T
1642  {
1643  using S = decltype(storage);
1644 
1645  if constexpr (JsonString<S>)
1646  {
1647  if constexpr (fly::SameAs<T, json_string_type>)
1648  {
1649  return storage;

1651  else
1652  {

1655  return *(JsonStringType::convert<T>(storage));
