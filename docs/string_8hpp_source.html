579 template <typename StringType>
580 std::vector<StringType>
581 BasicString<StringType>::split(view_type input, char_type delimiter, size_type count)
582 {
583  std::vector<StringType> elements;
584  StringType item;
585 
586  size_type start = 0;
587  size_type end = input.find(delimiter);
588 
589  auto push_item = [&elements, &count, &delimiter](view_type str)
590  {
591  if (!str.empty())
592  {
593  if ((count > 0) && (elements.size() == count))
594  {
595  elements.back() += delimiter;
596  elements.back() += str;
597  }
598  else
599  {
600  elements.push_back(StringType(str));
601  }
602  }
603  };
604 
605  while (end != StringType::npos)
606  {
607  item = input.substr(start, end - start);
608  push_item(item);
609 
610  start = end + 1;
611  end = input.find(delimiter, start);
612  }
613 
614  item = input.substr(start, end);
615  push_item(item);
616 
617  return elements;
618 }

671 bool BasicString<StringType>::wildcard_match(view_type source, view_type search)
672 {
673  static constexpr char_type s_wildcard = '*';
674  bool result = !search.empty();
675 
676  const std::vector<StringType> segments = split(search, s_wildcard);
677  size_type index = 0;
678 
679  if (!segments.empty())
680  {
681  if (result && (search.front() != s_wildcard))
682  {
683  result = source.starts_with(segments.front());
684  }
685  if (result && (search.back() != s_wildcard))
686  {
687  result = source.ends_with(segments.back());
688  }
689 
690  for (auto it = segments.begin(); result && (it != segments.end()); ++it)
691  {
692  index = source.find(*it, index);

696  result = false;
697  }
698  }
