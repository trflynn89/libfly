587 template <typename StringType>
588 std::vector<StringType>
589 BasicString<StringType>::split(view_type input, char_type delimiter, size_type count)
590 {
591  std::vector<StringType> elements;
592  StringType item;
593 
594  size_type start = 0;
595  size_type end = input.find(delimiter);
596 
597  auto push_item = [&elements, &count, &delimiter](view_type str)
598  {
599  if (!str.empty())
600  {
601  if ((count > 0) && (elements.size() == count))
602  {
603  elements.back() += delimiter;
604  elements.back() += str;
605  }
606  else
607  {
608  elements.push_back(StringType(str));
609  }
610  }
611  };
612 
613  while (end != StringType::npos)
614  {
615  item = input.substr(start, end - start);
616  push_item(item);
617 
618  start = end + 1;
619  end = input.find(delimiter, start);
620  }
621 
622  item = input.substr(start, end);
623  push_item(item);
624 
625  return elements;
626 }

680 {

684  const std::vector<StringType> segments = split(search, s_wildcard);
685  size_type index = 0;

687  if (!segments.empty())

852 template <typename StringType>
853 template <typename... ParameterTypes>
854 StringType BasicString<StringType>::format(
855  FormatString<ParameterTypes...> &&fmt,
856  ParameterTypes &&...parameters)
857 {
858  using FormatContext =
859  detail::BasicFormatContext<std::back_insert_iterator<StringType>, char_type>;
860 
861  if (fmt.has_error())
862  {
863  return format(FLY_ARR(char_type, "Ignored invalid formatter: {}"), fmt.error());
864  }
865 
866  const view_type view = fmt.view();
867 
868  StringType formatted;
869  formatted.reserve(view.size() * 2);
870 
871  auto params =
872  detail::make_format_parameters<FormatContext>(std::forward<ParameterTypes>(parameters)...);
873  FormatContext context(std::back_inserter(formatted), params);
874 
875  auto format_value = [&context](auto value)
876  {
877  using T = std::remove_cvref_t<decltype(value)>;
878 
879  if constexpr (std::is_same_v<T, detail::UserDefinedValue<FormatContext>>)
880  {
881  value.m_format(value.m_value, context);
882  }
883  else if constexpr (std::is_same_v<T, detail::StringValue<FormatContext>>)
884  {
885  value.m_format(value.m_value, value.m_size, context);
886  }
887  else
888  {
889  Formatter<T, char_type>().format(value, context);
890  }
891  };
892 
893  for (std::size_t pos = 0; pos < view.size();)
894  {
895  switch (const auto &ch = view[pos])
896  {
897  case s_left_brace:
898  if (view[pos + 1] == s_left_brace)
899  {
900  *context.out()++ = ch;
901  pos += 2;
902  break;
903  }
904 
905  context.spec() = *std::move(fmt.next_specifier());
906  context.arg(context.spec().m_position).visit(format_value);
907 
908  pos += context.spec().m_size;
909  break;
910 
911  case s_right_brace:
912  *context.out()++ = ch;
913  pos += 2;
914  break;
915 
916  default:
917  *context.out()++ = ch;
918  ++pos;
919  break;
920  }
921  }
922 
923  return formatted;
924 }
