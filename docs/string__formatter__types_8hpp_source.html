800 template <typename StringType, typename... ParameterTypes>
801 template <std::size_t N>
802 FLY_CONSTEVAL BasicFormatString<StringType, ParameterTypes...>::BasicFormatString(
803  const char_type (&format)[N]) noexcept :
804  m_lexer(format)
805 {
806  std::optional<char_type> ch;
807 
808  if constexpr (!(BasicFormatTraits::is_formattable_v<ParameterTypes> && ...))
809  {
810  on_error("An overloaded operator<< must be defined for all format parameters");
811  }
812 
813  while (!has_error() && (ch = m_lexer.consume()))
814  {
815  if (ch == s_left_brace)
816  {
817  if (m_lexer.consume_if(s_left_brace))
818  {
819  continue;
820  }
821  else if (m_specifier_count >= m_specifiers.size())
822  {
823  on_error("Exceeded maximum allowed number of specifiers");
824  }
825  else if (auto specifier = parse_specifier(SpecifierType::Full); specifier)
826  {
827  m_specifiers[m_specifier_count++] = *std::move(specifier);
828  }
829  }
830  else if (ch == s_right_brace)
831  {
832  if (m_lexer.consume_if(s_right_brace))
833  {
834  continue;
835  }
836 
837  on_error("Closing brace } must be esacped");
838  }
839  }
840 }

876 //==================================================================================================
877 template <typename StringType, typename... ParameterTypes>
878 constexpr auto BasicFormatString<StringType, ParameterTypes...>::parse_specifier(SpecifierType type)
879  -> std::optional<FormatSpecifier>
880 {
881  // The opening { will have already been consumed, so the starting position is one less.
882  const auto starting_position = m_lexer.position() - 1;
883 
884  FormatSpecifier specifier {};
885  parse_position(specifier);
886 
887  if ((type == SpecifierType::Full) && m_lexer.consume_if(s_colon))
888  {
889  parse_fill_and_alignment(specifier);
890  parse_sign(specifier);
891  parse_alternate_form_and_zero_padding(specifier);
892  parse_width(specifier);
893  parse_precision(specifier);
894  parse_locale_specific_form(specifier);
895  parse_type(specifier);
896  }
897  else
898  {
899  infer_type(specifier);
900  }
901 
902  if (!m_lexer.consume_if(s_right_brace))
903  {
904  on_error("Detected unclosed format string - must end with }");
905  return std::nullopt;
906  }
907  else if (!validate_specifier(specifier))
908  {
909  return std::nullopt;
910  }
911 
912  specifier.m_size = m_lexer.position() - starting_position;
913  return specifier;
914 }

933 //==================================================================================================
934 template <typename StringType, typename... ParameterTypes>
935 constexpr void BasicFormatString<StringType, ParameterTypes...>::parse_fill_and_alignment(
936  FormatSpecifier &specifier)
937 {
938  if (auto next = m_lexer.peek(1); next)
939  {
940  if ((*next == s_less_than_sign) || (*next == s_greater_than_sign) || (*next == s_caret))
941  {
942  specifier.m_fill = m_lexer.consume().value();
943  }
944  }
945 
946  if (m_lexer.consume_if(s_less_than_sign))
947  {
948  specifier.m_alignment = FormatSpecifier::Alignment::Left;
949  }
950  else if (m_lexer.consume_if(s_greater_than_sign))
951  {
952  specifier.m_alignment = FormatSpecifier::Alignment::Right;
953  }
954  else if (m_lexer.consume_if(s_caret))
955  {
956  specifier.m_alignment = FormatSpecifier::Alignment::Center;
957  }
958 }
